15  Eigen::Quaternion<T> expq(const Eigen::Quaternion<T>& q)
16  {
17  T a = q.vec().norm();
18  T exp_w = exp(q.w());
19 
20  if (a == T(0))
21  {
22  return Eigen::Quaternion<T>(exp_w, 0, 0, 0);
23  }
24 
25  Eigen::Quaternion<T> res;
26  res.w() = exp_w * T(cos(a));
27  res.vec() = exp_w * T(sinc(a)) * q.vec();
28 
29  return res;
30  }

32  template <typename T>
33  Eigen::Quaternion<T> logq(const Eigen::Quaternion<T>& q)
34  {
35  T exp_w = q.norm();
36  T w = log(exp_w);
37  T a = acos(q.w() / exp_w);
38 
39  if (a == T(0))
40  {
41  return Eigen::Quaternion<T>(w, T(0), T(0), T(0));
42  }
43 
44  Eigen::Quaternion<T> res;
45  res.w() = w;
46  res.vec() = q.vec() / exp_w / (sin(a) / a);
47 
48  return res;
49  }

167  template <typename T>
168  DualQuaternion<T> DualQuaternion<T>::inverse(void) const
169  {
170  T sqrLen0 = m_real.squaredNorm();
171  T sqrLenE = 2.0 * (m_real.coeffs().dot(m_dual.coeffs()));
172 
173  if (sqrLen0 > 0.0)
174  {
175  T invSqrLen0 = 1.0 / sqrLen0;
176  T invSqrLenE = -sqrLenE / (sqrLen0 * sqrLen0);
177 
178  DualQuaternion<T> conj = conjugate();
179  conj.m_real.coeffs() = invSqrLen0 * conj.m_real.coeffs();
180  conj.m_dual.coeffs() = invSqrLen0 * conj.m_dual.coeffs() + invSqrLenE * conj.m_real.coeffs();
181 
182  return conj;
183  }
184  else
185  {
186  return DualQuaternion<T>::zeros();
187  }
188  }

208  template <typename T>
209  void DualQuaternion<T>::normalize(void)
210  {
211  T length = m_real.norm();
212  T lengthSqr = m_real.squaredNorm();
213 
214  // real part is of unit length
215  m_real.coeffs() /= length;
216 
217  // real and dual parts are orthogonal
218  m_dual.coeffs() /= length;
219  m_dual.coeffs() -= (m_real.coeffs().dot(m_dual.coeffs()) * lengthSqr) * m_real.coeffs();
220  }
